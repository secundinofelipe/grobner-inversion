\begin{frame}[t]{A Conjectura Jacobiana}
\small
\begin{teobox}{Conjectura Jacobiana (Keller, 1939)}
Seja $\KK$ algebricamente fechado e $F\colon\KK^n\to\KK^n$ polinomial.
Se $\det(J_F) \in \KK^*$ , então $F$ é automorfismo polinomial.
\end{teobox}
