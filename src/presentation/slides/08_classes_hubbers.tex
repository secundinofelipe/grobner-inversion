\begin{frame}{A redução de Hubbers}
    Em 1994, Engelbert Hubbers obteve importantes reduções da Conjectura Jacobiana, mostrando que o problema geral pode ser transformado em oito casos particulares, seguindo o teorema:
\vspace{1cm}
    \begin{teobox}{Classificação polinomial de E.Hubbers}
            Seja $F = Id_{\KK^n} + H\colon \kk^4\to\kk^4$ uma aplicação polinomial, com $\det J_F \equiv 1$. Então existe uma transformação linear $T\colon \kk^4\to\kk^4$, tal que $T^{-1}\circ F\circ T$ assume uma das formas:
    \end{teobox}
\end{frame}

\begin{frame}{Classificação de Hubbers}

            \begin{block}{Caso 1}
                $F = (x_1, x_2, x_3, F_4)$ \\
                \vspace{0.2em}
                $\begin{aligned}
                    F_4 &= x_4 - a_4 x_1^3 - b_4 x_1^2 x_2 - c_4 x_1^2 x_3 - e_4 x_1 x_2^2 - f_4 x_1 x_2 x_3 \\&- h_4 x_1 x_3^2 - k_4 x_2^3 - l_4 x_2^2 x_3 - n_4 x_2 x_3^2 - q_4 x_3^3
                \end{aligned}$
            \end{block}
\vspace{0.9cm}
\begin{block}{Caso 3 ($c_1 \neq 0$)}
                $F = (x_1, F_2, x_3, F_4)$ \\
                \tiny
                $\begin{aligned}
                    F_2 &= x_2 - \tfrac{1}{3} x_1^3 - c_1 x_1^2 x_4 + 3 c_1 x_1 x_2 x_3 
                    - \tfrac{16 q_4 c_1^2 - r_4^2}{48 c_1^2} x_1 x_3^2 - \tfrac{1}{2} r_4 x_1 x_3 x_4 \\
                    &+ \tfrac{3}{4} r_4 x_2 x_3^2 - \tfrac{r_4 q_4}{12 c_1} x_3^3 - \tfrac{r_4^2}{16 c_1} x_3^2 x_4 \\
                    F_4 &= x_4 - x_1^2 x_3 + \tfrac{r_4}{4 c_1} x_1 x_3^2 - 3 c_1 x_1 x_3 x_4 
                    + 9 c_1 x_2 x_3^2 - q_4 x_3^3 - \tfrac{3}{4} r_4 x_3^2 x_4
                \end{aligned}$
            \end{block}

        \end{column}

        \begin{column}{0.48\textwidth}
            \begin{block}{Caso 2}
                $F = (x_1, F_2, x_3, F_4)$ \\
                \vspace{0.2em}
                $\begin{aligned}
                    F_2 &= x_2 - \tfrac{1}{3} x_1^3 - h_2 x_1 x_3^2 - q_2 x_3^3 \\
                    F_4 &= x_4 - x_1^2 x_3 - h_4 x_1 x_3^2 - q_4 x_3^3
                \end{aligned}$
            \end{block}
            \vspace{0.9cm}
            \begin{block}{Caso 4}

                $F = (x_1, x_2 - \tfrac{1}{3} x_1^3, F_3, F_4)$ \\
                \vspace{0.2em}
                $\begin{aligned}
                    F_3 &= x_3 - x_1^2 x_2 - e_3 x_1 x_2^2 - k_3 x_2^3 \quad
                    F_4 = x_4 - e_4 x_1 x_2^2 - k_4 x_2^3
                \end{aligned}$
            \end{block}
        \end{column}
